import pygame import sys # Initialize Pygame pygame.init() # Screen dimensions SCREEN_WIDTH, SCREEN_HEIGHT = 800, 600 screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Simple Shooting Game") # Colors WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) # Clock clock = pygame.time.Clock() # Player setup player = pygame.Rect(100, 250, 50, 50) player_speed = 5 # Enemy setup enemy = pygame.Rect(600, 250, 50, 50) enemy_alive = True # Bullet setup bullet_width, bullet_height = 10, 5 bullets = [] bullet_speed = 10 # Game loop running = True while running: screen.fill(WHITE) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Shooting bullets if event.type == pygame.KEYDOWN and event.key == pygame.K_SPACE: bullets.append(pygame.Rect(player.x + player.width, player.y + player.height // 2 - bullet_height // 2, bullet_width, bullet_height)) # Player movement keys = pygame.key.get_pressed() if keys[pygame.K_UP] and player.top > 0: player.y -= player_speed if keys[pygame.K_DOWN] and player.bottom < SCREEN_HEIGHT: player.y += player_speed # Update bullets for bullet in bullets[:]: bullet.x += bullet_speed if bullet.x > SCREEN_WIDTH: bullets.remove(bullet) # Check for collision with enemy if enemy_alive: for bullet in bullets: if enemy.colliderect(bullet): enemy_alive = False bullets.remove(bullet) # Draw player pygame.draw.rect(screen, BLACK, player) # Draw enemy if enemy_alive: pygame.draw.rect(screen, RED, enemy) # Draw bullets for bullet in bullets: pygame.draw.rect(screen, BLACK, bullet) # Refresh screen pygame.display.flip() clock.tick(60) pygame.quit() sys.exit()
